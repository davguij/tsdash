# Module: isEqual

Compares two objects and returns whether their values are equivalent

import { isEqual } from 'utilist';

isEqual({ key: 'value' }, { key: 'value' }); // returns true
isEqual({ key: 'value' }, { anotherKey: 'another value' }); // returns false
